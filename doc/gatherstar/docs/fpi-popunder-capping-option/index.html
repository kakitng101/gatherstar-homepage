Manage capping for Popunders and Fullpage Interstitials

While creating or editing a Popunder, Desktop/Mobile Fullpage Interstitial zone, Publishers have an option to enable/disable capping for their ad zone. With this option, Publishers may decide to manage capping (display frequency) on their side.

Supported Ad Formats: Popunder, Mobile Popunder, Desktop Fullpage Interstitial, Mobile Fullpage Interstitial.

Setting Up

Step 1: Enable/Disable capping through the snippet

Disable the capping on the Popunder snippet. The capping_enabled or ad_capping_enabled needs to be set to false in the ad snippet

<script type="application/javascript">
function() {
    //version 1.0.0
    var adConfig = {
    "ads_host": "a.example.com",
    "syndication_host": "s.example.com",
    "idzone": XXX, // Replace XXX with zone id
    "popup_fallback": false,
    "popup_force": false,
    "chrome_enabled": true,
    "new_tab": false,
    "frequency_period": 720,
    "frequency_count": 1,
    "trigger_method": 3,
    "trigger_class": "",
    "trigger_delay": 0,
    "capping_enabled": false, // set this to false
    "only_inline": false
};
}

<script type="application/javascript">
    var ad_idzone = "XXX", // Replace XXX with zone id
    ad_popup_fallback = false,
    ad_popup_force = false,
    ad_chrome_enabled = true,
    ad_new_tab = false,
    ad_frequency_period = 720,
    ad_frequency_count = 1,
    ad_trigger_method = 3,
    ad_trigger_delay = 0,
    ad_capping_enabled = false; // set this to false
</script>
<script type="application/javascript" src="https://a.example.com/popunder1000.js"></script>

Note: You need to complete Step 2 and Step 3 to ensure the ad is not triggered on every click.

Step 2: Setting up Event Listener

The event listener needs to be set up for creativeDisplayed event to be able to store/increment capping counter on publisher side

<script type="application/javascript">
        document.addEventListener(
            'creativeDisplayed-XXX', // Replace XXX with zone id
            function(e) {
                //Code to increment the display counter goes here. Below is just and example
                 const name = 'popunder-capping';
                 let match = document.cookie.match(new RegExp('(^| )' + name + '=([^;]+)'));
                 let counter = match ? parseInt(match[2], 10) : 0;
                 ++counter;
                 document.cookie = name + "=" + counter + "; path=/";
            }
        );
    </script>
